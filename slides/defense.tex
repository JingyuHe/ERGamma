\section{Exponential Reciprocal Gamma (ERG) Distribution}
\begin{frame}{ERG(0) Distribution}
\begin{definition}
	Random variable $X_0$ has an exponential reciprocal gamma distribution, ERG$(0)$, if and only if its density function $p_0(x)$ satisfies the following identity
	\begin{equation*}
	E\left( e^{-s^2 X_0}\right) = \int_{0}^\infty e^{- s^2x} p_0(x) d x = \frac{e^{-\gamma s}}{\Gamma(1+s)} , \; \; s > 0 .
	\end{equation*}
	where $\gamma = -\psi(1) \approx 0.57721$ is the Euler-Mascheroni constant and $\psi(s) = \frac{d}{ds}\log\Gamma(s)$ is the digamma function. 
\end{definition}
\begin{remark}
	\label{rmk:remark_1}
	The product representation for the reciprocal gamma function due to Weierstrass is,
	$$
	\frac{e^{-\gamma s}}{\Gamma(1+s)} = \prod_{k=1}^\infty \left(1 + \frac{s}{k}\right) e^{-\frac{s}{k}}, \, s \in \mathbb{C}\slash \{0, -1, -2, ...\}
	$$
\end{remark}
\end{frame}

\begin{frame}{Some distribution results in literature}
Hartman (1976) proves the existence of a distribution $V_t$, for $0 < t \leq e^{-\gamma}$, such that 
\begin{equation*}
E\left(e^{-s^2V_t}\right) = \frac{t^s}{\Gamma(1+s)}.
\end{equation*}
Roynette and Yor (2005) prove the existence of an infinitely divisible distribution $H_a^\Gamma $, with density function  $  p_a^\Gamma  ( x )$, such that for $a>0$
$$
E\left( e^{-\frac{1}{2}s^2 H_a^\Gamma}\right) = \int_0^\infty e^{-\frac{1}{2} s^2 x} p_a^\Gamma  ( x ) dx =\frac{\Gamma(a)}{\Gamma(a+s)} e^{\psi(a)s} 
$$
which follows from the general product representation for the reciprocal gamma function, 
\begin{equation*}
\frac{\Gamma(a)}{\Gamma\left(a + s\right)} e^{\psi(a)s} =\prod_{k=0}^{\infty}\left( 1 + \frac{s}{a+ k} \right)e^{-\frac{s}{a+k}}.
\end{equation*}
ERG(0) is a special case with $a = 1$. $\text{ERG}(0) \overset{d}{=} \frac{1}{2} H_1^\Gamma.$
\end{frame}

\begin{frame}{General ERG(c) Distribution}
	\begin{definition}
		\label{def:PG_dc}
		The ERG$(c)$ distribution is constructed as an exponential tilting of the ERG$(0)$ density. Its density function is
		\begin{equation*}
		p_c\left(x\right) = Z_c \cdot \exp\left(- c^2 x\right) p_0(x),\;  x, c >0.
		\end{equation*}
		Similarly, the integral identity of ERG$(c)$ is given by
		\begin{align*}
		E\left(e^{-s^2 X_c}\right) &= \prod_{k=1}^\infty \left( \frac{k + \sqrt{s^2 + c^2}}{k + c} \right) e^{-\frac{\sqrt{s^2 + c^2}-c}{k}}\\
		&= \frac{\Gamma(1+c)}{\Gamma(1+\sqrt{s^2+c^2})} e^{-\gamma(\sqrt{s^2+c^2}-c)}
		\end{align*}
	\end{definition}
\end{frame}

\begin{frame}{Relation to Generalized Inverse Gaussian}
\begin{theorem}
	\label{thm:theorem_1}
	The ERG$(c)$ class of distributions can be constructed as an infinite sum of independent generalized inverse Gaussian (GIG) distributions as follows
	$$
	\text{ERG}(c) \overset{d}{=} \sum_{k=1}^\infty \text{GIG}\left(-\frac{3}{2}, 2c^2, \frac{1}{2k^2}\right).
	$$ 
	In particular, when $c=0$, the GIG distribution reduces to inverse gamma distribution. Hence,
	\begin{align*}
	\text{ERG}(0) \overset{d}{=} \sum_{k=1}^\infty \frac{1}{4k^2} \Gamma^{-1}_k.
	\end{align*}
	where $\Gamma^{-1}_k$ are i.i.d. inverse gamma random variables with shape $\frac{3}{2}$ and scale 1. 
\end{theorem}
\end{frame}

\begin{frame}{Mean and Variance}
\begin{theorem}
	\label{thm:moments}
	If $G_k$ is a GIG random variable with $p = -\frac{3}{2}, a=2c^2, b=\frac{1}{2k^2}$, then the mean and variance of the tail infinite sum $\sum_{k=N}^\infty G_k$ are
	\begin{align*}
	E\left(\sum_{k=N}^\infty G_k\right) &= \frac{1}{2c}\left(\psi(N+c) - \psi(N)\right) \\
	Var\left(\sum_{k=N}^\infty G_k\right) &= \frac{1}{4c^3} \left(\psi(N+c) - \psi(N) - c \psi'(N+c)\right).
	\end{align*}
	Setting $N=1$ gives us the first two moments of ERG($c$).
\end{theorem}
\footnotesize
This theorem is useful for the generation ERG variables, construction of EM algorithm as well as approximation of Gibbs sampler.
\end{frame}



\section{MCMC and Data Augmentation}

\begin{frame}{Data Augmentation}
The idea of the data augmentation strategy is to introduce a group of auxiliary random variables $\bm{\omega} = (\omega_1, ..., \omega_n)^T$ such that for $\bm{\omega} \in \Omega$,
$$
p(x) = \int_\Omega p(x,\bm{\omega}) d\bm{\omega}
$$ 
and the joint density $p(x,\bm{\omega})$ after augmentation is easier to deal with, as it doesn't consist of gamma functions any longer.
\vspace{0.1in}
\footnotesize
\begin{table}[]
\centering
\begin{tabular}{@{}lcc@{}}
	\toprule
	& Integral Representation  &  Auxiliary Variables\\ \midrule
	$\Gamma(\alpha)$ & $\int_{0}^{\infty} x^{\alpha-1} e^{-x} dx$& Gamma  \\
	& & \\
	$\frac{1}{\Gamma(\alpha)}$& $\int_{0}^\infty \alpha e^{-\alpha^2x+\gamma\alpha} p_0(x) dx$& ERG\\
	& &\\
	$\frac{\Gamma(\alpha)}{\Gamma(\alpha+\beta)}$ & $\int_{0}^1 \frac{1}{\Gamma(\beta)} x^{\alpha-1}(1-x)^{\beta-1}dx$& Beta\\
	\bottomrule
\end{tabular}
\end{table}
\end{frame}

\begin{frame}{Goal}
We unify the (posterior) densities of many disparate gamma problems
\vspace{0.05in}
{
	\footnotesize
	\begin{equation*}
	p(x)  \propto {\color{red}\left(\prod_{\ell=1}^L \Gamma(g_\ell(x)) \right)}{\color{blue} \left(\prod_{m=1}^M \frac{1}{\Gamma(h_m(x))}\right)}{\color{green} \left(\prod_{n=1}^N\frac{\Gamma(j_n(x))}{\Gamma(j_n(x) + \beta_n)}\right)} \cdot x^{p-1} e^{-ax^2 + bx}
	\end{equation*}
}
\vspace{0.05in}
\begin{itemize}
	\item The support is $(0, \infty)$.
	\item $g,h,j$ are increasing linear functions. 
	\item $M\geq 1$ and $a \geq 0$ and assume the density is integrable.
	\item $g,h,j,\beta$ can depend on data.
\end{itemize}
\vspace{0.05in}
\bcol{By exploiting the integral identities related to gamma function, we derive a general data augmentation strategy for the above density. }
\end{frame}


\begin{frame}
\footnotesize
\begin{align*}
p(x)  &= C \cdot{\color{red}\left(\prod_{\ell=1}^L \Gamma(g_\ell(x)) \right)}{\color{blue} \left(\prod_{m=1}^M \frac{1}{\Gamma(h_m(x))}\right)}{\color{green} \left(\prod_{n=1}^N\frac{\Gamma(j_n(x))}{\Gamma(j_n(x) + \beta_n)}\right)} \cdot x^{p-1} e^{-ax^2 + bx}\\
&=\int C\cdot  G(x, \bm{\tau})\cdot H(x, \bm{\omega}) \cdot J(x, \bm{\eta}) \cdot Q(x) \cdot e^{-ax^2 + bx} d\bm{\tau}d\bm{\omega}d\bm{\eta} 
\end{align*}
where 
\begin{align*}
G(x, \bm{\tau}) &= \prod_{\ell=1}^L \tau_{\ell}^{g_{\ell}(x)-1} e^{-\tau_\ell} \\
H(x, \bm{\omega}) &= \exp\left\{-\sum_{m=1}^M h_m(x)^2 \omega_m + \gamma\sum_{m=1}^M h_m(x)\right\} \prod_{m=1}^M p_0(\omega_m) \\
J(x, \bm{\eta}) &= \prod_{n=1}^N \eta_n^{j_n(x)-1} (1-\eta_n)^{\beta_n-1}\\
Q(x) &= x^{p-1} \prod_{m=1}^M h_m(x).
\end{align*}
\end{frame}

\begin{frame}
	We may generalize the previous result in a few ways:
	\vspace{0.1in}
	\begin{enumerate}
		\item For multivariate $x$ of dimension $d$, if $g,h,j$ are all linear functions mapping from $\mathbb{R}^{+, d}$ to $\mathbb{R}^+$, then the data augmentation is still valid.
		\item The $x^p$ term can be replaced with a polynomial function of $x$, as long as it's always positive for $x > 0$.
		\item If the density $p(x)$ has extra factors which are not included in the general form , the strategy still works as long as we can find integral representations for those extra factors. (for example, P\'olya-Gamma in negative binomial regression)
	\end{enumerate} 
\end{frame}


\begin{frame}
\footnotesize
\begin{definition}
	The power truncated normal distribution, PTN$(p,a,b)$, has density function
	\begin{equation*}
	p(x) = C\cdot x^{p-1} e^{-ax^2+bx}, x,p,a>0
	\end{equation*}
\end{definition}
\bcol{\bf Gibbs Sampler:}
\begin{align*}
\tau_{\ell} | x, \bm{\tau}_{-\ell}, \bm{\omega,\eta} &\sim \Gamma(g_\ell(x),1), \; \ell = 1, 2, ..., L \\
\omega_m | x, \bm{\tau}, \bm{\omega}_{-m}, \bm{\eta} &\sim \text{ERG}(h_m(x)), \; m = 1, 2, ..., M\\
\eta_n | x, \bm{\tau}, \bm{\omega}, \bm{\eta}_{-n} &\sim \text{Beta}(j_n(x), \beta_n), \; n = 1, 2, ..., N\\
x | \bm{\tau}, \bm{\omega}, \bm{\eta} &\sim \sum_{m=0}^M \pi_k \cdot \text{PTN}(p+M, \tilde{a}, \tilde{b})
\end{align*}
$\{\pi_m\}_{m=0}^M$ are proportional to the coefficients in the polynomial $Q(x)$.
\begin{align*}
\tilde{a} &= a + \sum_{m=1}^M \omega_m\cdot(h_m'(0))^2,\\
\tilde{b} &= b + \left(\sum_{\ell=1}^{L} g_\ell'(0)\cdot\log\tau_{\ell}\right) + \left(\sum_{m=1}^M h_m'(0)(\gamma -2\omega_m h_m(0))\right) + \left(\sum_{n=1}^N j_n'(0)\cdot\log\eta_n\right). 
\end{align*}

\end{frame}

\begin{frame}
\text{In summary:}
\begin{itemize}
	\item ${\color{red}\left(\prod_{\ell=1}^L \Gamma(g_\ell(x)) \right)}$ results in $L$ gamma variables
	\item ${\color{blue} \left(\prod_{m=1}^M \frac{1}{\Gamma(h_m(x))}\right)}$ results in $M$ ERG variables.
	\item ${\color{green}\left(\prod_{n=1}^N\frac{\Gamma(j_n(x))}{\Gamma(j_n(x) + \beta_n)}\right)}$ results in $N$ Beta variables. 
	\item The parameters of auxiliary variables depends on $g,h,j,\beta$.
	\item Conditioned $x$, all auxiliary variables are independent. 
	\item Conditioned on auxiliary variables, $x$ has the density $x^{\tilde{p}-1}e^{-\tilde{a}x^2 + \tilde{b} x}$. 
	\item In many applications, $g,h,j$ are relatively simple. Observing that $\tilde{a}$ and $\tilde{b}$ are both the sums of numerous terms, we may use normal variables to approximate them, so that the sampling procedure for $(\bm{\tau,\omega,\eta})$ can be skipped.
\end{itemize}
\end{frame}


\begin{frame}{EM Algorithm}
\footnotesize
\begin{itemize}
	\item \textbf{\bcol{Expectation Step:}}\\
	\begin{align*}
	C(x | x_{(t)}) &:= E_{\bm{\tau,\omega, \eta}|x_{(t)}}\left[\log p(x, \bm{\tau,\omega, \eta})\right] \\
	=& \left(\sum_{\ell=1}^L g_{\ell}(x) \psi(g_\ell(x_{(t)})) \right) + \left(\sum_{m=1}^M \gamma h_m(x) - h_m(x)^2 \frac{\psi(1+h_m(x_{(t)})) + \gamma}{2h_m(x_{(t)})} \right) \\
	&+ \left(\sum_{n=1}^N j_n(x)\psi(j_n(x_{(t)})) - \psi(j_n(x_{(t)}) + \beta_n) \right) + \log Q(x) -ax^2 + bx
	\end{align*}
	Given the linearity of $g, h, j$ functions, it can be further simplified as 
	\begin{equation*}
	C(x | x_{(t)}) = \log Q(x) - \kappa_1 x^2 + \kappa_2 x
	\end{equation*}
	for some constant $\kappa_1 > 0$ and $\kappa_2 \in \mathbb{R}$, depending on $x_{(t)}$.
	\item \textbf{\bcol{Maximization Step:}}\\
	Since $C(x | x_{(t)}) \rightarrow -\infty$ as $x \rightarrow \infty$,  we conclude that $C(x | x_{(t)})$ as a function of $x$ has a maximizer on $(0, \infty)$, which can be found numerically. Furthermore, when $h_m(0) = 0$ for all $m$, the unique maximizer has a closed form
	\begin{equation*}\label{eqn:m-step}
	x^* := \arg\max_{x>0} C(x | x_{(t)}) = \frac{\kappa_2 + \sqrt{\kappa_2^2 + 8\kappa_1(p+M-1)}}{4\kappa_1}.
	\end{equation*}
\end{itemize}
\end{frame}


\section{Examples}

\begin{frame}{Gamma Shape Inference}
Gamma distribution
$$
p(x|\alpha, \beta) = \frac{\beta^\alpha}{\Gamma(\alpha)} x^{\alpha-1} e^{-\beta x}.
$$
GamCon distribution of type II (Damsleth, 1975), $\xi_2(\mu, \delta)$:
\begin{equation*}
p(x \mid \mu, \delta) = C_{\mu,\delta}\cdot {\color{red}\Gamma(\delta x + 1)} {\color{blue}\left(\prod_{m=1}^M \frac{1}{\Gamma(x)}\right)} e^{-\delta\log(\delta\mu) x}
\end{equation*}
Hence $L=1$ with $g_1(x) = \delta x+ 1$, $M=\delta$ with $h_m(x) = x$ and $N=0$. $p=1, a=0$ and $b = -\delta\log(\delta\mu)$.
\end{frame}

\begin{frame}
\begin{itemize}
	\item This is the conjugate prior for the shape parameter $\alpha$, when the scale parameter $\beta$ is unknown. $\beta | \alpha \sim \Gamma(\delta\alpha+1, \delta\eta)$.
	\item If the prior is $\xi_2(\eta/\mu,\delta)$, then the posterior is $\xi_2(\eta'/\mu',\delta')$, where
	\begin{align*}
	\delta' &= \delta + n,\\
	\eta' &= \frac{\delta}{\delta + n}\eta + \frac{n}{\delta + n} x_a, \\
	\mu' &= \mu^{\frac{\delta}{\delta + n}}\cdot x_g^{\frac{n}{\delta + n}}.
	\end{align*}
	$x_a$ and $x_g$ are the arithmetical and geometrical mean respectively.
	\item $\delta = 0$ means non-informative prior.
	\item We replicate Damsleth's examples with $n=5,10,30$. $\delta = 0$.
\end{itemize}
\end{frame}

\begin{frame}{Multinomial-Dirichlet Model}
The sampling model for the count vector $\bm{n}_s$ is then a multinomial distribution given probability vector $\bm{p}_s$,
\begin{equation*}
\bm{n}_s \mid N_s, \bm{p}_s  \sim \text{Multinomial}\left(N_s, \bm{p}_s\right).
\end{equation*}
The prior for $\bm{p}_s$ is 
\begin{align*}
\bm{p}_s \mid \bm{\alpha} \sim \text{Dirichlet}\left(\bm{\alpha}\right).
\end{align*}
The marginal likelihood $p\left(\bm{n}_s \mid \bm{\alpha}\right)$ is
\begin{equation*}
\prod_{s=1}^S p\left(\bm{n}_s \mid \bm{\alpha}\right) = \prod_{s=1}^S \left(\frac{\Gamma\left(\sum_{k=1}^K \alpha_k\right)}{\Gamma\left(\sum_{k=1}^K \left(n_{sk} + \alpha_k\right)\right)} \prod_{k=1}^K \frac{\Gamma\left(n_{sk} + \alpha_k\right)}{\Gamma\left(\alpha_k\right)}\right).
\end{equation*}
\end{frame}

\begin{frame}
We now reorganize the multivariate posterior density of $\bm\alpha$ as
\begin{equation*}
p(\bm{\alpha} \mid \bm{n}_1, \ldots, \bm{n}_S) = \left(\prod_{k=1}^K f(\alpha_k)\right){\color{green}\left(\prod_{s=1}^S \frac{\Gamma(\sum_k \alpha_k)}{\Gamma(\sum_k \alpha_k + N_s)}\right)} 
\end{equation*}
where 
\begin{equation*}\label{eqn:falpha}
f(\alpha_k) ={\color{red}\left(\prod_{s=1}^S\Gamma( \alpha_k + n_{sk})\right)} {\color{blue}\left(\prod_{s=1}^S \frac{1}{\Gamma(\alpha_k)}\right)}   \alpha_k^{a_0-1}e^{-b_0\alpha_k}.
\end{equation*}
The conditional posteriors are
\begin{align*}
\tau_{sk} \mid \bm{\alpha, \omega, \eta} &\sim \Gamma(\alpha_k+n_{sk}), \\
\omega_{sk} \mid \bm{\alpha, \tau, \eta} &\sim \text{ERG}(\alpha_k), \\
\eta_s \mid \bm{\alpha, \tau, \omega} &\sim \text{Beta}\left(\sum_{k=1}^K \alpha_k, N_s\right),\\
\alpha_k \mid \bm{\tau, \omega, \eta} &\sim \text{PTN}(S + a_0, a_k, b_k),
\end{align*}
where $a_k=\sum_{s=1}^S \omega_{sk}, b_k = S\gamma - b_0 + \sum_{s=1}^S\log(\tau_{sk}\cdot\eta_s)$. 
\end{frame}

\begin{frame}{Simulation Study}
\begin{itemize}
	\item (A) heterogeneous $\alpha_k = k/K$ and (B) homogeneous $\alpha_k = 1/K$.
	\item $a_0 = b_0/K$, so that the prior expectations are all equal to $1/K$.
	\item $N_s=500$ for $s=1,2, ..., S$.
	\item $S\in \{100, 1000\}, K\in \{10, 50\}$ and $b_0\in \{0.1, 1, 5\}$.
	\item 4 MCMC algorithms in comparison:
	\begin{itemize}
		\item DA: Gibbs sampler which iteratively samples $\bm{\tau, \omega, \eta}$ and $\bm\alpha$.
		\item DA-N: replace the parameters in the conditional posterior of $\bm\alpha$ with approximating normal variables and skip the sampling of $\bm{\tau, \omega, \eta}$ in DA.
		\item DA-E: replace the parameters in the conditional posterior of $\bm\alpha$ with corresponding expectations and skip the sampling of $\bm{\tau, \omega, \eta}$ in DA.
		\item MH: random-walk Metropolis-Hasting sampler.
	\end{itemize}
   \item For each combination of $(b_0, S, K)$, we run the simulation for 50 times (the first 200 samples are dropped each time).
\end{itemize}
\end{frame}

\begin{frame}
\bcol{Metrics:}
\begin{itemize}
	\item Root Mean Square Error: $$\text{RMSE} = \sqrt{\frac{1}{TK}\sum_{t=1}^{T} \Vert \bm{\alpha}^{(t)} - \bm{\alpha} \Vert^2 }.$$ $T=500$ is the posterior sample size. 
	\item Effective sample size ratio (ESSR): $$\text{ESSR} = \frac{1}{K}\sum_{k=1}^K\frac{\lambda_k^2}{\sigma_k^2},$$ which measures the serial correlation between posterior samples. $\lambda^2$ is the sample variance and $\sigma^2$ is the estimate of the spectral density at frequency zero. 
	\item Algorithm running time in seconds
\end{itemize}
\end{frame}

\begin{frame}{Discussion}
\begin{itemize}
	\item The class of ERG distributions are developed as mixing distributions for models with gamma functions. 
	This adds to the literature on normal variance-mean mixtures by showing that they extend to a wide class of applications. 
	\item Our ensuing data augmentation strategy facilitates full posterior inference for parameters in models which were hitherto hard to provide inference.
	\item We believe that the computational strategies developed here will provide the foundation for new and richly structured hierarchical gamma models. 
	\item Avenues for future research: regularized scale allocation models, multivariate ERG theory, etc.
\end{itemize}
\end{frame}
