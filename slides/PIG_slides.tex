\begin{frame}
	\frametitle{Motivation}
	\begin{itemize}[<+->]\setlength{\itemsep}{15pt}
		\item Classification and regression tree (CART)\\
		      Fully grown tree without penalty of complexity, then prune the tree.
		\item Random Forest\\
		      Implicit regularization, the number of variables and data used to grow each tree. Also tree-level regularization.
		\item Gradient boosting\\
		      Add regularization to the objective function.
		\item Bayesian addtive regression trees (BART)\\
		      An explicit regularization prior of tree model.
	\end{itemize}
\end{frame}
